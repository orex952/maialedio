## 例
```cpp example
#include <cassert>
#include <chrono>

namespace chrono = std::chrono;

int main()
{
  chrono::hh_mm_ss time{chrono::hours{15} + chrono::minutes{30} + chrono::seconds{10}};
  assert(chrono::is_am(time.hours()) == false);
  assert(chrono::is_pm(time.hours()) == true);
}
```
