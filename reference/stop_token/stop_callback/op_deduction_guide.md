## 例
```cpp example
#include <stop_token>
#include <type_traits>

struct X {
  void operator()() {}
};

int main()
{
  X x;
  std::stop_token st;
  std::stop_callback cb { st, x };

  static_assert(std::is_same_v<decltype(cb)::callback_type, X>);
}
```
