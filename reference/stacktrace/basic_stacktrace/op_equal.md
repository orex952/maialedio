## 例

#include <stacktrace>

void g() {
  std::stacktrace a = std::stacktrace::current();
  std::stacktrace b = std::stacktrace::current();
  std::stacktrace c = a;

  assert(a == c);
  assert(a != b);
}

void f() {
  g();
}

  f();
}
